[docs]    def get_time_slices(self):

        # If no time slices then we just use the month parameter

        if not self.time_slices:
            yield self.months

        # If a time slice parameter was provided then we use this to set the month parameter
        else:
            if TimeSlice.FULL in self.time_slices:
                pass

            elif TimeSlice.YEAR in self.time_slices:
                pass

            elif TimeSlice.SEASON in self.time_slices:
                for season in Season:
                    yield MONTHS_BY_SEASON[season]

            elif TimeSlice.QUARTER in self.time_slices:
                for quarter in Quarter:
                    yield MONTHS_BY_QUARTER[quarter]

            elif TimeSlice.MONTH in self.time_slices:
                for month in Month:
                    yield [month]



[docs]    def get_summary_task_items(self):

        from itertools import product

        for (acq_min, acq_max), months in product(self.get_epochs(), self.get_time_slices()):
            yield acq_min, acq_max, months

[docs]    def run(self):

        self.setup_arguments()
        self.process_arguments(self.parser.parse_args())
        self.log_arguments()

        if self.local_scheduler:
            luigi.build(self.create_summary_tasks(), local_scheduler=self.local_scheduler, workers=self.workers)

        else:
            import luigi.contrib.mpi as mpi
            mpi.run(self.create_summary_tasks())

[docs]    def complete(self):
        from luigi.task import flatten

        for output in flatten(self.output()):
            if not output.exists():
                return False

        for dep in flatten(self.deps()):
            if not dep.complete():
                return False

        return True

[docs]    def get_tiles_from_csv(self):

        if os.path.isfile(self.get_tile_csv_filename()):
            with open(self.get_tile_csv_filename(), "rb") as f:
                import csv

                reader = csv.DictReader(f)
                for record in reader:
                    _log.debug("Found CSV record [%s]", record)
                    yield Tile.from_csv_record(record)

[docs]    def get_tiles_from_db(self):

        from datacube.api.query import list_tiles

        x_list = [self.x]
        y_list = [self.y]

        for tile in list_tiles(x=x_list, y=y_list, acq_min=self.acq_min, acq_max=self.acq_max,
                               satellites=[satellite for satellite in self.satellites],
                               include=[month for month in self.months],
                               dataset_types=self.get_dataset_types()):
            yield tile
